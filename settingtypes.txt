
[General]

# Set to false to disable all built-in items and their recipes
crafttools.enable_items (Enable Built-In Items) bool true

# Set to false to disable the built-in recipes for all tools using items from the 'default' mod of MTG
crafttools.enable_default_recipes (Enable Default Recipes) bool true

# Set to false to disable all the hand tools and their recipes
crafttools.enable_hand_tools (Enable Built-In Hand Tools) bool true

# Set to false to disable all the consumables and their recipes
crafttools.enable_consumables (Enable Built-In Reusable Consumables) bool true

# Set to false to disable playing sounds when a tool breaks
crafttools.enable_break_sounds (Enable Tool Breaking Sounds) bool true

# Set to false to disable playing sounds when using tools in a recipe
crafttools.enable_craft_sounds (Enable Tool Crafting Sounds) bool true
